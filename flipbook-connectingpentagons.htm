  2. drawGeom[v_, ratio_, n_] := Module[{ring},
   ring[c_, r_, depth_] := Module[{ps},
     ps = c + r {Cos[#], Sin[#]} & /@ (2. Pi Range[v]/v);

     If[depth == 0, Polygon[ps],
      ring[(c + #)/2, r ratio, depth - 1] & /@ ps]];

   Graphics[{Transparent,
     EdgeForm[{Opacity[.28], Black}],
     ring[0., 1., n]}]];
