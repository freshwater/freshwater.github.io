        EdgeForm[Opacity[.3]],
        (*original*)EdgeForm[Purple], Purple, h[{#1, #2, 0} & @@@ pts],
        (*riemann*)EdgeForm[Blue], Blue, h[tmp = toRiemann[pts]],
        (*riemann inverse*)EdgeForm[Red], Red, h[tmp = {#1, #2, -#3} & @@@ tmp],

    Axes -> None, PlotRange -> All]];
